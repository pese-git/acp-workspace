# 2.1 — Устранение двойного кэша сессий

**Приоритет:** 🟠 Высокий  
**Оценка:** 1.5 дня  
**Файлы:** `src/codelab/server/protocol/core.py`, `src/codelab/server/storage/`

---

## Проблема

В системе два независимых кэша сессий:

1. `ACPProtocol._sessions: dict[str, SessionState]` — in-memory словарь, per-connection экземпляр.
2. `JsonFileStorage._cache: dict[str, SessionState]` — второй кэш внутри хранилища.

**Сценарий расхождения:**
1. Клиент подключается, создаёт сессию → данные идут в `_sessions` и `JsonFileStorage._cache`.
2. Сервер перезапускается → `_sessions` очищается, `JsonFileStorage._cache` тоже (новый процесс).
3. Новый клиент вызывает `session/prompt` с существующим `sessionId`.
4. `session/prompt` ищет сессию только в `self._sessions` — там пусто → ошибка "session not found".
5. `_hydrate_session_cache_from_storage()` вызывается только из `session/list` — не помогает.

Дополнительно: `_hydrate_session_cache_from_storage()` загружает **все** сессии в память при каждом вызове `session/list` — потенциальный OOM.

---

## Решение

Убрать `_sessions` как отдельный кэш из `ACPProtocol`. Вместо этого — LRU-кэш внутри `Storage`.

### Шаг 1 — Добавить LRU-кэш в `SessionStorage`

Создать `src/codelab/server/storage/cached.py`:

```python
"""LRU-обёртка над SessionStorage с ограниченным кэшем."""

from __future__ import annotations

from collections import OrderedDict
from typing import Any

import structlog

from .base import SessionStorage
from ..protocol.state import SessionState

logger = structlog.get_logger()

DEFAULT_CACHE_SIZE = 200  # максимум сессий в памяти


class CachedSessionStorage(SessionStorage):
    """SessionStorage с LRU-кэшем фиксированного размера.

    Оборачивает любой SessionStorage backend и добавляет кэш
    для снижения нагрузки на I/O при частых обращениях к активным сессиям.
    """

    def __init__(
        self,
        backend: SessionStorage,
        max_size: int = DEFAULT_CACHE_SIZE,
    ) -> None:
        self._backend = backend
        self._max_size = max_size
        # OrderedDict работает как LRU: move_to_end при каждом обращении
        self._cache: OrderedDict[str, SessionState] = OrderedDict()

    def _put(self, session: SessionState) -> None:
        """Добавить в кэш, вытолкнув самую старую запись при переполнении."""
        session_id = session.session_id
        if session_id in self._cache:
            self._cache.move_to_end(session_id)
        else:
            if len(self._cache) >= self._max_size:
                evicted_id, _ = self._cache.popitem(last=False)
                logger.debug("session_cache_evicted", session_id=evicted_id)
            self._cache[session_id] = session

    def _invalidate(self, session_id: str) -> None:
        self._cache.pop(session_id, None)

    async def save_session(self, session: SessionState) -> None:
        await self._backend.save_session(session)
        self._put(session)

    async def load_session(self, session_id: str) -> SessionState | None:
        if session_id in self._cache:
            self._cache.move_to_end(session_id)
            return self._cache[session_id]
        session = await self._backend.load_session(session_id)
        if session is not None:
            self._put(session)
        return session

    async def delete_session(self, session_id: str) -> bool:
        self._invalidate(session_id)
        return await self._backend.delete_session(session_id)

    async def list_sessions(
        self,
        cwd: str | None = None,
        cursor: str | None = None,
        limit: int = 100,
    ) -> tuple[list[SessionState], str | None]:
        # list_sessions не прогревает кэш — только чтение через backend
        return await self._backend.list_sessions(cwd=cwd, cursor=cursor, limit=limit)

    async def session_exists(self, session_id: str) -> bool:
        if session_id in self._cache:
            return True
        return await self._backend.session_exists(session_id)

    @property
    def cache_size(self) -> int:
        return len(self._cache)
```

### Шаг 2 — Удалить `_sessions` из `ACPProtocol`

```python
# core.py — удалить:
```

### Шаг 3 — Обновить обращения к сессиям в handlers

Вместо `sessions[session_id]` везде использовать `await storage.load_session(session_id)`.

Создать вспомогательный метод в `ACPProtocol`:

```python
async def _require_session(
    self, session_id: str, request_id: JsonRpcId
) -> SessionState | ProtocolOutcome:
    """Загрузить сессию или вернуть ошибку."""
    session = await self._storage.load_session(session_id)
    if session is None:
        return ProtocolOutcome(
            response=ACPMessage.error_response(
                request_id,
                code=-32002,
                message=f"Session not found: {session_id}",
            )
        )
    return session
```

### Шаг 4 — Обновить `_hydrate_session_cache_from_storage`

Метод `session/list` теперь просто делегирует в Storage без загрузки всего в `_sessions`:

```python
# БЫЛО:
async def _hydrate_session_cache_from_storage(self) -> None:
    cursor = None
    while True:
        page, next_cursor = await self._storage.list_sessions(cursor=cursor, limit=50)
        for session_state in page:
            self._sessions[session_state.session_id] = session_state  # OOM риск
        if next_cursor is None:
            break
        cursor = next_cursor

# СТАЛО:
# Метод больше не нужен — удалить.
# session/list теперь:
async def _handle_session_list(self, message: ACPMessage) -> ProtocolOutcome:
    params = message.params or {}
    cursor = params.get("cursor")
    page, next_cursor = await self._storage.list_sessions(cursor=cursor, limit=self._session_list_page_size)
    return ProtocolOutcome(response=session.build_session_list_response(message.id, page, next_cursor))
```

## Тесты

```python
@pytest.mark.asyncio
async def test_session_survives_protocol_restart():
    """Сессия должна быть доступна после пересоздания ACPProtocol."""
    storage = InMemoryStorage()
    protocol1 = ACPProtocol(storage=storage)

    # Создать сессию
    await protocol1.handle(ACPMessage.request("initialize", {...}))
    result = await protocol1.handle(ACPMessage.request("session/new", {"cwd": "/tmp"}))
    session_id = result.response.result["sessionId"]

    # "Перезапустить" — создать новый экземпляр протокола с тем же storage
    protocol2 = ACPProtocol(storage=storage)

    # Сессия должна быть найдена
    load_result = await protocol2.handle(
        ACPMessage.request("session/load", {"sessionId": session_id})
    )
    assert load_result.response.result is not None
    assert "error" not in load_result.response.result


def test_lru_cache_eviction():
    from codelab.server.storage.cached import CachedSessionStorage
    from codelab.server.storage.memory import InMemoryStorage

    backend = InMemoryStorage()
    cache = CachedSessionStorage(backend=backend, max_size=2)

    # Заполнить кэш до лимита
    s1 = SessionState(session_id="s1", cwd="/", mcp_servers=[])
    s2 = SessionState(session_id="s2", cwd="/", mcp_servers=[])
    s3 = SessionState(session_id="s3", cwd="/", mcp_servers=[])

    import asyncio
    asyncio.run(cache.save_session(s1))
    asyncio.run(cache.save_session(s2))

    assert cache.cache_size == 2

    # Добавить третью — вытолкнет первую (LRU)
    asyncio.run(cache.save_session(s3))
    assert cache.cache_size == 2
    assert "s1" not in cache._cache
    assert "s3" in cache._cache
```
